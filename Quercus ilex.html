2. Database and variables

Database used in the models: A list with with eighteen dataframes, each with the variables for one of the eighteen tree species selected.

sp <- final_list_sp[[params$sp_vector]]

2.1. Explanatory variables

  • Species initial basal area (m2 ha-1) - ba.ha_sp - Calculated as the total basal area of the target species inside the plot extrapolated to the hectare.

  • Stand density (no. trees ha-1) - density.ha - Calculated as the number of trees inside the plot extrapolated to the Ha.

  • Stand development (mm) - mean_dbh_large - Stand development. Calculated as the mean diameter of the larger trees inside the plot extrapolated to the hectare.

  • Harvest regime (%) - ba_perc_HARVEST - An integration of the intensity and frequency of harvest in the area, at the grid level.

  • Climate moisture index (kg m-2 month-1) - cmi_clim - Mean climate moisture index (CMI) in the plot.

  • VPD anomaly (Pa) - vpd_gs_anomaly - Vapour pressure deficit (VPD) anomaly for the growing season (March - August) in the plot.

  • Country (Belgium, Czech Republic, France, Netherlands, Poland, Spain, Sweden) - country - Country in which the plot is found.

2.2. Response variables

  • Basal area gains (m2 ha-1 yr-1) - ba.gain_yearcm - Basal area gains due to growth + ingrowth of the target species inside the plot in between measurements.

  • Basal area losses (m2 ha-1 yr-1) - ba.loss_yearcm - Basal area losses due to mortality of the target species trees inside the plot in between measurements.

  • Mortality (0/1) - mortality - Presence/absence of a mortality event (1 or more trees dead) of the target species inside the plot in between measurements.

  • Basal area changes (m2 ha-1 yr-1) -ba.change.dif_yearcm - Change in basal area of the target species between measurements. Calculated as the difference between the final basal area and the initial basal area of the target species inside the plot.

- Cross-validation

The cross validation was calculated to evaluate de predictive ability of the models.

# This is the number of repetitions that the test sub sample will be validated with the trained model
n_repes <- 10
# Train is a random subsample of 80% of the data
cv <- modelr::crossv_mc(sp, test = .2, n = n_repes)

crossval_g <- function(x){ 
  # x is the number of the repetition (n_repes)
  
  ### Select train and test #
  # train just for fit the model
  train <- as_tibble(cv$train[[x]])
  # test to predict 
  test <- as_tibble(cv$test[[x]]) 
  
  ### Error: country levels on train-test are different #
  # in some cases, it depends on the random subsample. 
  # I filter by the common countries in both datasets. 
  # Only a few rows are lost
  common_country <- 
    intersect(unique(train$country), unique(test$country))
  
  test_filtered <- test |> 
    filter(country %in% common_country)
  
  train_filtered <- train |> 
    filter(country %in% common_country) 
  
  # model with train
  gam.model <- 
    gam(log(ba.gain_yearcm) ~ 
        log(ba.ha_spcm) + 
          log(density.ha) + 
          ba_perc_HARVEST + 
          country +
          te(cmi_clim, vpd_gs_anomaly, mean_dbh_large, bs = "cr"),
        family = "gaussian",
        data = train_filtered,
        method = "REML")
  
  # results with modelr
  results <- tibble(
    mean_gains = mean(log(train_filtered$ba.gain_yearcm)),
    # Higher, better:
    r2 = modelr::rsquare(gam.model, data = test_filtered),
    # Lower, better:
    rmse = modelr::rmse(gam.model, data = test_filtered),
    nrmse = rmse / mean_gains, 
    # Lower, better
    mae = modelr::mae(gam.model, data = test_filtered),
    nmae = mae / mean_gains,
    # To see the consequences of the country filter
    test_row = nrow(test_filtered), 
    train_row = nrow(train_filtered)
    
  )
  
  return(results)
}

# Run SAFELY. Evita que la iteracin se pare con cualquier fallo

- Cross-validation

The cross validation was calculated to evaluate de predictive ability of the models.

## Number of repetitions #
# This is the number of repetitions that the test 
# sub sample will be validated with the trained model
n_repes <- 10

## Split the data #
# Train is a random subsample of 80% of the data
cv <- modelr::crossv_mc(sp, test = .2, n = n_repes)

crossval_b <- function(x){ 
  # x is the number of the repetition (n_repes)
  
  ### Select train and test #
  # train just for fit the model
  train <- as_tibble(cv$train[[x]])
  # test to predict 
  test <- as_tibble(cv$test[[x]]) 
  
  ### Error: country levels on train-test are different #
  # in some cases, it depends on the random subsample. 
  # I filter by the common countries in both datasets. 
  # Only a few rows are lost
  common_country <- 
    intersect(unique(train$country), unique(test$country))
  
  test_filtered <- test |> 
    filter(country %in% common_country) |> 
    mutate(mortality = as.factor(mortality))
  
  train_filtered <- train |> 
    filter(country %in% common_country) |> 
    mutate(mortality = as.factor(mortality))
  
  ### Model with train #
  binomial.losses <-
    gam(
      mortality ~
        ba.ha_spcm +
        ba_perc_HARVEST +
        country +
        te(cmi_clim, vpd_gs_anomaly, mean_dbh_large, bs = "cr"),
      #weights = density.ha,
      family = "binomial",
      data = train_filtered,
      method = "REML"
    )
  
  ### Predictions with test #
  predictions <- 
    predict(binomial.losses, 
            type = "response",
            newdata = test_filtered, 
    ) 
  
  ### Results accuracy #
  # create a confusion matrix with thershold value = 0.5
  results_acc <- 
    tibble(
      obs = test_filtered$mortality,
      preds = case_when(
        predictions >= 0.5 ~ 1, 
        T ~ 0
      )
    ) |> 
    mutate(across(everything(), as.factor))
  
  # calculate accuracy and kappa with caret package
  conf <- caret::confusionMatrix(
    data = results_acc$obs,
    reference = results_acc$preds,
    positive = "0")
  
  ### Results
  results <- tibble(
    accuracy = conf[["overall"]][["Accuracy"]],
    kappa = conf[["overall"]][["Kappa"]],
    test_row = nrow(test_filtered), 
    train_row = nrow(train_filtered)
    
  )
  
  return(results)
  
  
}

- Cross-validation

The cross validation was calculated to evaluate de predictive ability of the models.

# This is the number of repetitions that the test sub sample will be validated with the trained model
n_repes <- 10
# Train is a random subsample of 80% of the data
cv <- modelr::crossv_mc(sp, test = .2, n = n_repes)

crossval_c <- function(x){ 
  # x is the number of the repetition (n_repes)
  
  ### Select train and test #
  # train just for fit the model
  train <- as_tibble(cv$train[[x]])
  # test to predict 
  test <- as_tibble(cv$test[[x]]) 
  
  ### Error: country levels on train-test are different #
  # in some cases, it depends on the random subsample. 
  # I filter by the common countries in both datasets. 
  # Only a few rows are lost
  common_country <- 
    intersect(unique(train$country), unique(test$country))
  
  test_filtered <- test |> 
    filter(country %in% common_country)
  
  train_filtered <- train |> 
    filter(country %in% common_country) 
  
  # model with train
  gam.model <- #change
      gam(ba.change.dif_yearcm ~
            ba.ha_spcm + 
            density.ha + 
            ba_perc_HARVEST + 
            country +
            te(cmi_clim, vpd_gs_anomaly, mean_dbh_large, bs = "cr"),
          family = "gaussian",
          data = train_filtered, # !
          method = "REML")
  
  # results with modelr
  results <- tibble(
    mean_change = mean(train_filtered$ba.change.dif_yearcm, 
                      na.rm = TRUE),
    # Higher, better:
    r2 = modelr::rsquare(gam.model, data = test_filtered),
    # Lower, better:
    rmse = modelr::rmse(gam.model, data = test_filtered),
    nrmse = rmse / mean_change, 
    # Lower, better
    mae = modelr::mae(gam.model, data = test_filtered),
    nmae = mae / mean_change,
    # To see the consequences of the country filter
    test_row = nrow(test_filtered), 
    train_row = nrow(train_filtered)
    
  )
  
  return(results)
  
}

# Function to create the map for each demographic variable.  

create_maps <- function(var_name, var_title) {
  
  # Keeping just the values inside quantiles 5 and 95 to leave out outlayers
  
  q5 <- quantile(sp[var_name], probs = 0.05, na.rm = T)
  q95 <- quantile(sp[var_name], probs = 0.95, na.rm = T)
  
  sp_data1 <- sp[!(sp[var_name] < q5),] 
  sp_data <- sp_data1[!(sp_data1[var_name] > q95),]
  
  # Base Europe map 
  border <- map_data(map = "world")
  
  border2 <- border |>
    mutate(coords = paste(long, lat, sep = " ")) |>
    group_by(group, region) |>
    dplyr::summarize(polygon = str_c(coords, collapse = ", "))
  
  create_vect <- function(group_n) {
    vect(paste0("POLYGON ((", border2[border2$group == group_n, "polygon"], "))"),
         crs = "EPSG:4326")
  }
  
  border_spat <-
    map(seq_along(seq_along(border2$group)), \(x) create_vect(x))
  
  border_spat2 <- vect(
    x = border_spat,
    crs = "EPSG:4326"
  )
  
  # Data 
  dat_df<- terra::vect(
    x = sp_data,
    geom = c("longitude", "latitude"),
    crs = "EPSG:4326"
  )
  
  dat_df_df <- as.data.frame(dat_df)
  col_data <- as.numeric(dat_df_df[[var_name]])
  
  # Map 
  
  plot <- ggplot() +
    geom_spatvector(
      data = border_spat2,
      fill = "grey90",
      color = "grey90"
    ) +
    geom_spatvector(
      data = dat_df,
      aes(color = col_data),
      size = .05, 
      alpha = .7
    ) +
    scale_color_viridis_c(
      bquote(.(var_title) ~ paste("(", m^2, ha^-1, yr^-1,")"))
    ) +
    coord_sf(
      xlim = c(-15, 37.5),
      ylim = c(35, 75),
      expand = FALSE
    ) +
    labs(
        title = var_title
    ) +    
    theme(
      panel.background = element_rect(fill = "white"),
      panel.grid.major = element_line(color = "grey80"),
      axis.text = element_text(size = 6), 
      title = element_text(size = 8), 
      axis.title = element_text(size = 7), 
      plot.title = element_text(size = 6)
    )
  
  return(plot)
}
